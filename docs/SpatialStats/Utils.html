def self.fdr(pvals, alpha)
  n = pvals.size
  b_bound = alpha / n
  pvals.sort!

  p_val = b_bound
  (0..n - 1).each do |i|
    p_fdr = (i + 1) * b_bound
    break unless pvals[i] <= p_fdr

    p_val = p_fdr
  end
  p_val
end
